118  void _newSlogan(Game& game) {
119  std::ifstream inFile("resource/slogans.txt");
120  int numLines = 0;
121  int chosenLine = 0;
122  std::string string = "";
123  // Count the number of slogans in file
124  while ( std::getline(inFile, string) ) {
125  ++numLines;
126  }
127  // Choose random line
128  srand(time(nullptr));
129  chosenLine = rand() % (numLines + 1);
130  // Clear the eof flag and move cursor to start of file
131  inFile.clear();
132  inFile.seekg(0, std::ios::beg);
133  // Read until chosenLine
134  for (int i = 0; i <=chosenLine; i++) {
135  std::getline(inFile, string);
136  }
137  // Update the line to _slogan
138  _slogan.setString(string);
139  // Update the _slogan position, center and 150px from bottom
140  _slogan.setPosition(_resolution.x/2 - _slogan.getGlobalBounds().width/2, _resolution.y - _resolution.y * 0.21);
141  // Update the timestamp
